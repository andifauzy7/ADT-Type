	WHILE( gerak->nb != NULL AND test!=TRUE) DO
	 		IF ( gerak -> fs != NULL )
				THEN	push_stack(Head,gerak)
					gerak <- gerak->fs.
					JML <- JML + 1
					IF (JML > JML_TEMP)
						THEN JML_TEMP <- JML
					ENDIF
			ENDIF
			IF ( isi_stack AND gerak->nb == NULL AND gerak->fs == NULL)
				THEN	gerak <- pop_stack(Head)
					JML <- JML - 1
					IF(gerak==ujung) THEN test <- TRUE
			ENDIF

	ENDWHILE
